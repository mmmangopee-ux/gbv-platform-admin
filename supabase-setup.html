Database Setup

Connect your platform to Supabase

Step 1: Create Database Tables

Run this SQL in your Supabase SQL editor:

-- Create cases table
CREATE TABLE cases (
id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
case_code TEXT UNIQUE NOT NULL,
institution_id TEXT,
portal_type TEXT DEFAULT 'public',
status TEXT DEFAULT 'new',
location TEXT,
incident_type TEXT,
description TEXT,
contact_info TEXT,
created_at TIMESTAMP DEFAULT NOW()
);

-- Create professionals table
CREATE TABLE professionals (
id UUID DEFAULT gen_random_uuid() PRIMARY KEY,
email TEXT UNIQUE NOT NULL,
institution_id TEXT,
portal_type TEXT,
created_at TIMESTAMP DEFAULT NOW()
);

-- Enable Row Level Security
ALTER TABLE cases ENABLE ROW LEVEL SECURITY;
ALTER TABLE professionals ENABLE ROW LEVEL SECURITY;

-- Create policies
CREATE POLICY "Anyone can insert cases" ON cases FOR INSERT WITH CHECK (true);
CREATE POLICY "Professionals can view all cases" ON cases FOR SELECT USING (true);
